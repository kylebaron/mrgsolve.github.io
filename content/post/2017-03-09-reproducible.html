--- title: Reproducible results with set.seed() date: '2017-03-09' --- 
library(mrgsolve)
library(dplyr)

Demo model

code <- '
$PKMODEL cmt="CENT"
$PARAM CL = 1, V = 20

$OMEGA 1 1

$TABLE
capture ETA1 = ETA(1);
capture ETA2 = ETA(2);
'
mod <- mcode("test", code)

Single dose

set.seed(9911)
single <- mod %>% ev(amt=100) %>% mrgsim(nid=100, end=24)

Multi dose

set.seed(9911)
multi <- mod %>% ev(amt=100, ii=24, addl=9) %>% mrgsim(nid=100, end=240)
